具体代码如下

package main

import (
	"fmt"
	"unsafe"
)

func main() {
	var (
		q bool
		n byte
		g uint8
		b int8
		c int16
		h uint16
		m rune
		d int32
		i uint32
		o float32
		k int
		l uint
		e int64
		j uint64
		p float64
		s complex64
		r string
		t complex128
	)

	fmt.Println("bool       = ", unsafe.Sizeof(q))
	fmt.Println("byte       = ", unsafe.Sizeof(n))
	fmt.Println("uint8      = ", unsafe.Sizeof(g))
	fmt.Println("int8       = ", unsafe.Sizeof(b))
	fmt.Println("int16      = ", unsafe.Sizeof(c))
	fmt.Println("uint16     = ", unsafe.Sizeof(h))
	fmt.Println("rune       = ", unsafe.Sizeof(m))
	fmt.Println("int32      = ", unsafe.Sizeof(d))
	fmt.Println("uint32     = ", unsafe.Sizeof(i))
	fmt.Println("float32    = ", unsafe.Sizeof(o))
	fmt.Println("int        = ", unsafe.Sizeof(k))
	fmt.Println("uint       = ", unsafe.Sizeof(l))
	fmt.Println("int64      = ", unsafe.Sizeof(e))
	fmt.Println("uint64     = ", unsafe.Sizeof(j))
	fmt.Println("float64    = ", unsafe.Sizeof(p))
	fmt.Println("complex64  = ", unsafe.Sizeof(s))
	fmt.Println("string     = ", unsafe.Sizeof(r))
	fmt.Println("complex128 = ", unsafe.Sizeof(t))
}
